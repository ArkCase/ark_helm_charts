NOTE: Legacy deployments must be running on 0.7.78 or newer, and Modern deployments must be on 0.8.0-104 or newer

Alfresco:
    ARKCASE_ALFRESCO_URL
	ARKCASE_ALFRESCO_USERNAME
	ARKCASE_ALFRESCO_PASSWORD

	Legacy:
		secret=${arkcase}-core
			if (contentType != "alfresco") fail();
			contentApi - "/alfresco"
			adminUsername
			adminPassword

	Modern:
		secret=${arkcase}-content-main
			if (bucket != nil) fail();
			url - "/alfresco"
			username
			password

Database:
    ARKCASE_JDBC_DRIVER
    ARKCASE_JDBC_URL
    ARKCASE_JDBC_USERNAME
    ARKCASE_JDBC_PASSWORD

	Legacy:
		secret=${arkcase}-core
			dbDriver
			dbUrl
			dbUsername
			dbPassword

	Modern:
		secret=${arkcase}-rdbms-arkcase
			username
			password

		secret=${arkcase}-core-config
			jdbcDriver
			jdbcUrl

Solr:
	ARKCASE_SOLR_URL
	ARKCASE_SOLR_USERNAME
	ARKCASE_SOLR_PASSWORD

	Legacy:
		secret=${arkcase}-core
			solrScheme + solrHost + solrPort + "/solr"

	Modern:
		secret=${arkcase}-search-main
			url + "/solr"
			username (optional)
			password (optional)
